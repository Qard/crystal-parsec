    def pass_through(p : Parsec::Parser(B)) : Parsec::Parser(A) forall B #

    Sequence another parser after this parser, ignoring the result of the second one, and returning the result of this parser.

    px = one_or_more(digit)
  .pass_through(string "px")
  .map {|x| x.join() }
  .map {|x| x.to_i }
px.parse("10px").should eq 10
px.parse("10").class.should eq ParseError
